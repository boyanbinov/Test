function ternarySearch(l, r, key, ar) { if (r >= l) { let mid1 = l + parseInt((r - l) / 3, 10); let mid2 = r - parseInt((r - l) / 3, 10); if (ar[mid1] == key) { return mid1; } if (ar[mid2] == key) { return mid2; } if (key < ar[mid1]) { return ternarySearch(l, mid1 - 1, key, ar); } else if (key > ar[mid2]) { return ternarySearch(mid2 + 1, r, key, ar); } else { return ternarySearch(mid1 + 1, mid2 - 1, key, ar); } } return -1; }

При тристранното търсене масивът първо трябва да бъде сортиран. При тристранното търсене масивът е разделен на три части, като за да изчислим индекса на лявата част можем да използваме формулата mid1 = l + (r-l)/3, формулата за дясната част е mid2 = r – (r-l)/3. Първото действие на тристранното търсене е да провери дали търсеният елемент не съвпада с mid1, след това проверява и mid2. Ако търсеният елемент е по-малък от mid1 търсенето продължава в ляво от mid1, като се преизчисляват mid1 и mid2. Ако търсеният елемент е по-голям от mid2 търсенето продължава в дясно от mid2, като се преизчисляват mid1 и mid2. Ако търсеният елемент е по-голям от mid1 и по-малък от mid2, то търсеният елемент е между mid1 и mid2. Скоростта на тристранното търсене е O(log3 n).
