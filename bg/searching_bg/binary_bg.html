function binarySearch(arr, l, r, x){ if (r >= l) { let mid = l + Math.floor((r - l) / 2); if (arr[mid] == x) return mid; if (arr[mid] > x) return binarySearch(arr, l, mid - 1, x); return binarySearch(arr, mid + 1, r, x); } return -1; }

При двоичното търсене масивът първо трябва да бъде сортиран. Двоичното търсене започва търсенето си от средата на масива, като се сравнява стойността на търсения елемент и стойността на елемента в средата на масива. Ако търсената стойност е по-голяма от средния елемент, то търсенето продължава в дясната част на масива, като отново масива се дели наполовина. Ако търсената стойност е по-малка от средния елемент, то търсенето продължава в лявата част на масива. Тези действия се повтарят, докато търсеният елемент не бъде намерен.
Средната скорост на двоичното търсене е O(log n). В най-добрия случай скоростта е O(1), а в най-лошия случай скоростта е O(log n).
