function cocktailSort(arr) { let swapped = true; let start = 0; let end = arr.length; while (swapped == true) { swapped = false; for (let i = start; i < end - 1; ++i) { if (arr[i] > arr[i + 1]) { let temp = arr[i]; arr[i] = arr[i + 1]; arr[i + 1] = temp; swapped = true; } } if (swapped == false) break; swapped = false; end = end - 1; for (let i = end - 1; i >= start; i--) { if (arr[i] > arr[i + 1]) { let temp = arr[i]; arr[i] = arr[i + 1]; arr[i + 1] = temp; swapped = true; } } start = start + 1; } }

void CocktailSort(int a[], int n) { bool swapped = true; int start = 0; int end = n - 1; while (swapped) { swapped = false; for (int i = start; i < end; ++i) { if (a[i] > a[i + 1]) { swap(a[i], a[i + 1]); swapped = true; } } if (!swapped) break; swapped = false; --end; for (int i = end - 1; i >= start; --i) { if (a[i] > a[i + 1]) { swap(a[i], a[i + 1]); swapped = true; } } ++start; } }
Взимаме първият елемент на масива и го сравняваме със следващия и разменяме стойностите им, ако първият е по-голям от втория. След това сравняваме вторият елемент с третия и пак разменяме, ако има нужда. Когато обхождането стигне до края на масива, вместо да започне своето обхождане отначало, той се връща назад, като обхожда елементите отдясно наляво и отново разменя елементите, докато не достигне началото на масива. Горепосочените стъпки се повтарят, докато масивът не бъде сортиран. Средната му скорост е O(n2). В най-добрият случай, когато елементите са почти на правилните си места скоростта му се измерва в O(n). Тъй като средната скорост е O(n2), следователно и скоростта в най-лошия случай отново се измерва в O(n2).
