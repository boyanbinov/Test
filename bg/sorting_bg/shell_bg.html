public int shell_sort(int []arr) { int n = arr.Length; for (int gap = n/2; gap > 0; gap /= 2) { for (int i = gap; i < n; i += 1) { int temp = arr[i]; int j; for (j = i; j >= gap && arr[j - gap] > temp; j -= gap) arr[j] = arr[j - gap]; arr[j] = temp; } } return 0; }
int shellSort(int arr[], int n) { for (int gap = n/2; gap > 0; gap /= 2) { for (int i = gap; i < n; i += 1) { int temp = arr[i]; int j; for (j = i; j >= gap && arr[j - gap] > temp; j -= gap) arr[j] = arr[j - gap]; arr[j] = temp; } } return 0; }
Сортирането на Шел започва сравнението и разменянето на елементите, с онези, които са отдалечени, преди да се извърши сравнението със съседните. Първо определяме разстоянието (gap) между елементите, които трябва да бъдат сортирани. При първата итерация делим размера на масива на две и получаваме разстоянието между елементите. След това обхождането на масива продължава, докато не се стигне до края на масива, междувременно елементите, които са селектирани се сравняват и се поставят на правилното място. Когато първата итерация на масива приключи, разстоянието се намалява наполовина. Отново се обхожда масива и се правят необходимите сравнения и размени. Разстоянието намалява, докато не достигне едно, когато достигне разстояние едно, на практика сортирането става сортиране чрез вмъкване.
Средната скорост на сортировката не може да се определи, защото зависи от разстоянието между елементите.
